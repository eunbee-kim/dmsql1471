	<button id="removeBtn" data-oper='remove' class="btn btn-danger">삭제</button>

2. 삭제시에 "삭제하시겠습니까?" confirm처리하기.
     	var formObj=$("form");
    	$("#removeBtn").on("click",function(e){
    		e.preventDefault();
    		
    		var operation = $(this).data("oper");
    		
    		console.log(operation);
    		
    		if(operation==="remove"){
    			var confirm_val=confirm("삭제하시겠습니까?");
    			if(confirm_val==true){
    			formObj.attr("action","/board/remove").attr("method","post");;
    			}else {
    				return;
    			}
    		}
    		formObj.submit();
    	});
